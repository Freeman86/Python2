x1 = int(float("Введите первую координату X: "))
y1 = int(float("Введите первую координату Y: "))
x2 = int(float("Введите вторую координату X: "))
y2 = int(float("Введите вторую координату Y: "))

k = (y1 - y2) / (x1 - x2)
b = y2 - k*x2

if b < 0:
    print(f"y = {k} * x + {b}")
else:
    print(f"y = {k} * x - {b}")
